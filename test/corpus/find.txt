========
Find substring first implicit occurrence
========
FIND 'a' in str.
---
(source
    (find
        (substring_spec
            (string_literal))
        (identifier)))
========
Find substring first explicit occurrence
========
FIND FIRST OCCURRENCE OF 'a' in str.
---
(source
    (find
        (occurrence_spec)
        (substring_spec
            (string_literal))
        (identifier)))

(source

========
Find section 
========
FIND ALL OCCURRENCES OF foo 
    IN SECTION OFFSET 12 LENGTH 5 OF str.
---
(source
    (find
        (occurrence_spec)
        (substring_spec
            value: (identifier))
        (section
            off: (number)
            len: (number))
        dobj: (identifier)))

========
Find options meta information
========
FIND ALL OCCURRENCES OF 'know' IN text1 
        MATCH COUNT  DATA(cnt) 
        MATCH OFFSET DATA(off) 
        MATCH LENGTH DATA(len). 
---
(source
    (find
        (occurrence_spec)
        (substring_spec
            (string_literal))
        (identifier)
        (find_options
            (operation_count_spec
                (declaration_expression
                    (identifier)))
            (operation_offset_spec
                (declaration_expression
                    (identifier)))
            (operation_length_spec
                (declaration_expression
                    (identifier))))))

========
Find pcre submatches
========
FIND PCRE `(\w+)\W+\1\W+(\w+)\W+\2` 
     IN text 
     IGNORING CASE 
     MATCH OFFSET FINAL(moff) 
     MATCH LENGTH FINAL(mlen) 
     SUBMATCHES FINAL(s1) FINAL(s2).
---
(source
    (find
        (pcre_spec
            (string_literal))
        (identifier)
        (find_options
            (case_sensitivity_spec)
            (operation_offset_spec
                (declaration_expression
                    (identifier)))
            (operation_length_spec
                (declaration_expression
                    (identifier)))
            (submatches_spec
                (declaration_expression
                    (identifier))
                (declaration_expression
                    (identifier))))))
